    expug v0.9.0 Expug.Visitor View Source

    Internal helper for traversing an AST.

    iex> node = %{
...>   title: "Hello",
...>   children: [
...>     %{title: "fellow"},
...>     %{title: "humans"}
...>   ]
...> }
iex> Expug.Visitor.visit(node, fn node ->
...>   {:ok, Map.update(node, :title, ".", &(&1 <> "."))}
...> end)
%{
   title: "Hello.",
   children: [
     %{title: "fellow."},
     %{title: "humans."}
   ]
}
